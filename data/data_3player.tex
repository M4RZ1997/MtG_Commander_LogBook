	\DTLnewdbentry{LogBook3Player}{MulliganNumber}{#7}%
	\conditionalSet{\listMulliganCauses}{LogBook3Player}{MulliganCause}{#8}%
	\conditionalSet{\listObviousProblems}{LogBook3Player}{ObviousProblems}{#9}%
}
